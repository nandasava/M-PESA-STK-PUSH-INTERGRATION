// Example: Initiate M-Pesa STK Push
const stkPush = async (phone, amount, accountRef) => {
  const response = await fetch('https://sandbox.safaricom.co.ke/mpesa/stkpush/v1/processrequest', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${accessToken}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      BusinessShortCode: "174379",
      Password: Buffer.from("174379" + passkey + timestamp).toString("base64"),
      Timestamp: timestamp,
      TransactionType: "CustomerPayBillOnline",
      Amount: amount,
      PartyA: phone,
      PartyB: "174379",
      PhoneNumber: phone,
      CallBackURL: "https://yourdomain.com/callback",
      AccountReference: accountRef,
      TransactionDesc: "Payment for services"
    })
  });
  return await response.json();
};

PayPal Integration

Global payment processing with the world's most recognized platform:

  • Express Checkout integration
  • Subscription and recurring payments
  • Payment gateway for e-commerce
  • Mass payouts to multiple recipients
  • Invoicing and billing solutions
  • Fraud protection and security
// Example: Create PayPal payment
const createPayment = async (amount, description) => {
  const response = await fetch('https://api.paypal.com/v1/payments/payment', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${accessToken}`
    },
    body: JSON.stringify({
      intent: 'sale',
      payer: { payment_method: 'paypal' },
      transactions: [{
        amount: { total: amount, currency: 'USD' },
        description: description
      }],
      redirect_urls: {
        return_url: 'https://yourdomain.com/success',
        cancel_url: 'https://yourdomain.com/cancel'
      }
    })
  });
  return await response.json();
};
